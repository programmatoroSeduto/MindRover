CREATE DATABASE frog_studios_db;


CREATE TABLE credenziali_utenti (
	id_utente INT NOT NULL AUTO_INCREMENT,
	email VARCHAR(50) NOT NULL,
	password VARCHAR(255) NOT NULL,
	PRIMARY KEY (id_utente)
);


CREATE TABLE profili_utenti (
	id_utente INT NOT NULL,
	first_name VARCHAR(50) NOT NULL,
	last_name VARCHAR(50) NOT NULL,
	descrizione TEXT(5000),
	stato VARCHAR(255),
	nickname VARCHAR(50) NOT NULL,
	FOREIGN KEY (id_utente) REFERENCES credenziali_utenti(id_utente),
	PRIMARY KEY (id_utente)
);


CREATE TABLE articoli (
	id_articolo INT NOT NULL AUTO_INCREMENT,
	id_autore INT NOT NULL, 
	titolo VARCHAR(255) NOT NULL,
	sottotitolo VARCHAR(255),
	elenco_tag VARCHAR(255),
	contenuto_html LONGTEXT,
	FOREIGN KEY (id_autore) REFERENCES credenziali_utenti(id_utente),
	PRIMARY KEY(id_articolo)
);


CREATE TABLE donazioni (
	id_donazione INT NOT NULL AUTO_INCREMENT,
	somma_donazione FLOAT NOT NULL,
	id_utente INT NOT NULL,
	FOREIGN KEY (id_utente) REFERENCES credenziali_utenti(id_utente),
	PRIMARY KEY(id_donazione)
)
